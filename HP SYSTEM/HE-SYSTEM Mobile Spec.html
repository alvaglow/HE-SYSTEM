// src/api/client.ts — Axios with auto-refresh
import axios from 'axios';
import * as SecureStore from 'expo-secure-store';

const api = axios.create({
  baseURL: 'https://api.happy-english.edu.vn/v1',
  timeout: 10000,
});

api.interceptors.request.use(async (config) => {
  const token = await SecureStore.getItemAsync('accessToken');
  if (token) config.headers.Authorization = `Bearer ${token}`;
  return config;
});

api.interceptors.response.use(
  (res) => res,
  async (err) => {
    if (err.response?.status === 401) {
      // Attempt silent token refresh
      const refreshed = await refreshTokens();
      if (refreshed) return api(err.config);
      await logout(); // redirect to login
    }
    return Promise.reject(err);
  }
);

Push Notifications
APNs (iOS) and FCM (Android) via Expo Notifications
// src/hooks/useNotifications.ts
import * as Notifications from 'expo-notifications';
import { useEffect } from 'react';

Notifications.setNotificationHandler({
  handleNotification: async () => ({
    shouldShowAlert: true,
    shouldPlaySound: true,
    shouldSetBadge:  true,
  }),
});

export function useNotifications() {
  useEffect(() => {
    // Register device token with backend
    const register = async () => {
      const { status } = await Notifications.requestPermissionsAsync();
      if (status !== 'granted') return;
      const { data: token } = await Notifications.getExpoPushTokenAsync();
      await api.post('/users/push-token', { token });
    };
    register();

    // Handle notification taps → navigate to relevant screen
    const sub = Notifications.addNotificationResponseReceivedListener(
      response => {
        const { type, entityId } = response.notification.request.content.data;
        if (type === 'fee')        router.push('/student/fees');
        if (type === 'attendance') router.push('/student/attendance');
        if (type === 'grade')      router.push('/student/results');
        if (type === 'otp')        router.push('/student/attendance');
      }
    );
    return () => sub.remove();
  }, []);
}

Offline Strategy
What works without internet — React Query + AsyncStorage

✅ Available Offline

Dashboard (cached data) · Timetable (current week) · Attendance records (last 30 days) · Grades and results · Fee invoices · Messages (read-only) · Profile info · Academic calendar

🔌 Requires Connection

OTP generation & submission · Fee payment · Wallet top-up · Sending messages · Submitting grades · Leave applications · Push notification delivery · Real-time attendance

// React Query offline config — staleTime + gcTime
const queryClient = new QueryClient({
  defaultOptions: {
    queries: {
      staleTime: 5 * 60 * 1000,   // 5 min — don't refetch
      gcTime:    24 * 60 * 60 * 1000, // 24h — keep in cache
      retry: 2,
      networkMode: 'offlineFirst',   // serve from cache when offline
    },
  },
});

// Persist cache to AsyncStorage for app restart
import { persistQueryClient } from '@tanstack/query-persist-client-core';
import { createAsyncStoragePersister } from '@tanstack/query-async-storage-persister';
persistQueryClient({
  queryClient,
  persister: createAsyncStoragePersister({ storage: AsyncStorage }),
});
